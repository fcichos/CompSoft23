def draw(pa,canvas, color='white'):
    with hold_canvas(canvas):
        canvas.clear()
        canvas.fill_style = color
        canvas.global_alpha = 0.2
        for i in range(N):
            canvas.fill_arc(pa[i].position[0], pa[i].position[1], 3, 0, 2*np.pi)

class Particle():
    def __init__(self,position=np.array([0,0])):
        self.position=np.zeros(2)
        self.position[:]=position

    def update(self):
        dx=np.random.normal()
        dy=np.random.normal()
        tmp=np.zeros(2)
        tmp[:]=self.position
        tmp=tmp+np.array([dx,dy])
        self.position[:]=tmp

class RandomWalk(Thread):
    def __init__(self,canvas):
        self.canvas = canvas
        super(RandomWalk, self).__init__()

    def run(self):
        for _ in range(1000):
            for i in range(N):
                p[i].update()
            draw(p,self.canvas, 'black')

            sleep(0.03)
