current_time = time()

def move_particles(x, speed_x, y, speed_y):
    global current_time

    elapsed_time = time() - current_time

    x += np.array(speed_x * elapsed_time, dtype=np.int32)
    y += np.array(speed_y * elapsed_time, dtype=np.int32)

    current_time = time()
